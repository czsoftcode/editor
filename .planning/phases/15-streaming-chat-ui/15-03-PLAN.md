**2. Pridat migracni metodu na Settings:**

```rust
impl Settings {
    /// One-time migration from plugin settings (WASM era) to native AI settings.
    /// Called on startup. Only migrates if ai_settings_migrated is false
    /// and plugin settings contain Ollama configuration.
    pub fn migrate_plugin_ai_settings(&mut self) -> bool {
        if self.ai_settings_migrated {
            return false;
        }

        // Hledat Ollama plugin settings (typicky pod klicem "gemini" nebo "ollama")
        let mut migrated = false;
        for (_id, ps) in &self.plugins {
            if let Some(url) = ps.config.get("OLLAMA_URL") {
                if !url.is_empty() && self.ollama_base_url == default_ollama_url() {
                    self.ollama_base_url = url.clone();
                    migrated = true;
                }
            }
            if let Some(key) = ps.config.get("OLLAMA_API_KEY") {
                if !key.is_empty() && self.ollama_api_key.is_empty() {
                    self.ollama_api_key = key.clone();
                    migrated = true;
                }
            }
            if let Some(model) = ps.config.get("MODEL") {
                if !model.is_empty() && self.ai_default_model.is_empty() {
                    self.ai_default_model = model.clone();
                    migrated = true;
                }
            }
            // Prevzit expertise a reasoning depth
            if ps.expertise != AiExpertiseRole::default() && self.ai_expertise == AiExpertiseRole::default() {
                self.ai_expertise = ps.expertise;
                migrated = true;
            }
            if ps.reasoning_depth != AiReasoningDepth::default() && self.ai_reasoning_depth == AiReasoningDepth::default() {
                self.ai_reasoning_depth = ps.reasoning_depth;
                migrated = true;
            }
        }

        if migrated {
            self.ai_settings_migrated = true;
        }
        migrated
    }
}
```
